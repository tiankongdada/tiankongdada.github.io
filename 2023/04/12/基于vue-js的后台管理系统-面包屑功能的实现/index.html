基于vue.js的后台管理系统-面包屑功能的实现

发表于2023-04-12|更新于2023-04-12|前端教程
|字数总计:1.5k|阅读时长:7分钟|阅读量:

前言

从后台管理系统代码中理出来的单独的面包屑功能的代码，使用了element UI和vue-router。

效果

关于vue-router

matched是Vue Router中的一个路由记录属性，它指向当前路由成功匹配的所有路由记录的数组，包括父路由记录和子路由记录。

在Vue Router中，路由记录是路由匹配过程中生成的对象，包含了与当前路径匹配的路由信息，以及路由参数、查询参数、命名视图等相关信息。每个路由记录都可以通过一个route对象来访问。

当一个路由成功匹配时，其对应的路由记录会被添加到matched数组中，而其他路由记录则会被排除。例如：

const router = new VueRouter({
  routes: [
    {
      path: '/foo',
      component: Foo,
      children: [
        {
          path: 'bar',
          component: Bar
        }
      ]
    }
  ]
})

router.beforeEach((to, from, next) => {
  console.log(to.matched)  // 输出匹配的路由记录
  next()
})

在上面的例子中，当访问路径为/foo/bar时，to.matched将输出一个包含两个路由记录的数组，分别是父路由记录和子路由记录。其中，父路由记录包含了/foo路径的路由信息，而子路由记录包含了/foo/bar路径的路由信息。

需要注意的是，matched数组中的路由记录是按照嵌套层级从外到内排列的，因此可以通过遍历这个数组来访问所有路由记录的相关信息。

实现

项目结构：

views中是首页，电子数码及子页面，服装鞋帽及子页面。

路由的meta中，showInbreadcrumb用于判断是否显示在面包屑中，title为在面包屑显示的名称

代码

Router/index.js

import Vue  from "vue";
import VueRouter from "vue-router";


Vue.use(VueRouter)


//1.创建路由
//2.路由配置
//3.创建router实例，传入路由配置
//4.将路由挂载到根节点

const routes = [
    //匹配空路由，重定向到根路由
    {
        path:'',
        redirect: '/home',
        meta:{
            showInbreadcrumb:false
        }
    },
    //根路由
    {
        path:'/home',
        component: ()=>import('../views/home'),
        name:'home',
        meta:{
            title:"首页",
            showInbreadcrumb:true
        }
    },
    //电子数码
    {
        path:'/electronics',
        name:'电子数码',
        component: ()=>import('@/views/home/index.vue'),
        redirect: '/electronics/computer',
        meta:{
            title:"电子数码",
            showInbreadcrumb:true
        },
        children:[
            {
                path:'computer',
                name:'computer',
                component: ()=>import('@/views/electronics/computer/index.vue'),
                meta: {
                    title:"电脑",
                    showInbreadcrumb:true
                }
            },
            {
                path:'phone',
                name:'手机',
                component: ()=>import('@/views/electronics/phone/index.vue'),
                meta:{
                    title:"手机",
                    showInbreadcrumb:true
                }
            },
            {
                path:'tv',
                name:'电视',
                component: ()=>import('@/views/electronics/tv/index.vue'),
                meta:{
                    title:"电视",
                    showInbreadcrumb:true
                }
            }
        ]
    },
    //服装鞋帽
    {
        path:'/clothing',
        name:'服装鞋帽',
        component: ()=>import('@/views/home/index.vue'),
        redirect: '/clothing/tops',
        meta:{
            title:"服装鞋帽",
            showInbreadcrumb:true
        },
        children:[
            {
                path:'tops',
                name:'上装',
                component:  ()=>import('@/views/clothing/tops/index.vue'),
                meta:{

            },
            {
                path:'lower',
                name:'下装',

                meta:{
                    title:"鞋子",

                }
            }
        ]
    },
    //前面所有路由都没匹配到时，重定向到根路由
    {
        path:'*',
        redirect:'/',
        meta:{
            showInbreadcrumb:false
        } 
    },   
]

const router = new VueRouter({ 
    routes
})

export default router

home/index.vue

<template>
  <div class="home">
    <el-container>
      <el-aside width="200px">
        <!-- 侧边导航栏 -->
          <div class="sideMenu">
              <el-menu
                  default-active="0"
                  class="el-menu-vertical-demo"
                  active-text-color="#ffd04b"
                  >
                      <div v-for="(item,index) in routes" :key="index" :index="index+''">
                          <!-- 没有二级菜单的 -->
                          <el-menu-item :index="index+''" v-if="!item.children">
                              <router-link :to="{name:item.name}">{{item.meta.title}}</router-link>
                          </el-menu-item>
                          <!-- 有二级菜单的 -->
                          <el-submenu :index="index+''" v-else>
                              <template slot="title">{{item.meta.title}}</template>
                              <el-menu-item  v-for="(item_,index_) in item.children" :key="index_" :index="index+'-'+index_">
                                  <router-link :to="{name:item_.name}">{{item_.meta.title}}</router-link>
                              </el-menu-item>
                          </el-submenu>
                      </div>
              </el-menu>
          </div>
      </el-aside>
      <el-container>
        <el-header>
              <div class="content">
              <!-- 面包屑 -->
                <div class="breadcrumb">
                    <el-breadcrumb separator-class="el-icon-arrow-right">
                        <el-breadcrumb-item v-for="(item,index) in breadcrumb" :key="index" :to="{ path: item.path}">{{item.meta.title}}</el-breadcrumb-item>
                    </el-breadcrumb>
                </div>
              
              </div>
        </el-header>
        <el-main>
          <!-- 路由出口 -->
          <router-view></router-view>
        </el-main>
      </el-container>
    </el-container>
      
      
  </div>
</template>

<script>
export default {
  data(){
      return{
      }
  },
  computed:{
      // 侧边导航数据
      routes(){
          // 从$router.options中获取所有路由信息并过滤
          return this.$router.options.routes.filter((item)=>{
              return item.meta.showInbreadcrumb
          });
      },
      // 面包屑数据
      breadcrumb(){
          // 根据路由配置meta中的showInbreadcrumb字段过滤
          let matchedArr = this.$route.matched.filter((item)=>{
              return item.meta.showInbreadcrumb}
          );
          // 因为首页比较特殊，必须一直显示在面包屑第一个，如果没有首页路由信息，手动添加到最前面
          if(matchedArr[0].meta.title !== '首页'){
              matchedArr.unshift(
                  {
                      path:'/home',
                      meta:{  
                          title:"首页",
                          showInbreadcrumb:true
                      }
                  },
              )
          }
          return matchedArr;
      },
  }
}

</script>
